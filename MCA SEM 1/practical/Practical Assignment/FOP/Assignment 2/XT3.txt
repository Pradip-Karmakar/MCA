#include<stdio.h>
#include<conio.h>
void main()
{
	int rows,cols, matrix[10][10], j, i, upper = 1, lower = 1;
	printf("Enter Rows : ");
	scanf("%d",&rows);
	printf("Enter Cols : ");
	scanf("%d",&cols);
	if(rows != cols) 
	{
		printf("\nNo possible Matrix in these rows and cols\n");
	}
	else
	{
		for(i = 0; i < rows; i++) 
		{
			for(j = 0; j < cols; j++) 
			{
				printf("Enter (%d,%d) Element : ",i,j);
				scanf("%d",&matrix[i][j]);
			}
		}
		for(i = 0; i < rows; i++) 
		{
			for(j = 0; j < cols; j++) 
			{
				if(j < i && matrix[i][j] != 0)
				{
					upper=0;
				}
				if(j > i && matrix[i][j] != 0)
				{
					lower=0;
				}
			}
		}
		if(upper && lower)
		{
			printf("\nThis is Diagonal Matrix");
		}
		else if(upper)
		{
			printf("\nThis is Upper Triangular Matrix\n");
		}
		else if(lower) 
		{
			printf("\nThis is Lower Triangular Matrix\n");
		}
		else
		{
			printf("\n Not Matching Patterns in Matrix \n");
		}
	}
getch();
}
