{

	for(i = 0; i < limit; i++)
	{
		printf("Enter The %d String : ", i + 1);
		scanf("%s",name[i]);
	}
	
	for(j = 0; j < limit; j++)

		strcpy(temp,name[j]);

			token = strcmp(name[i],temp);
			if(token < 0)
			{
				strcpy(temp,name[i]);	
				k=i;
			}
		}

		strcpy(temp2,name[j]);
		strcpy(name[j],temp);
		strcpy(name[k],temp2);

getch();
