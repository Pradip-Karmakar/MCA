#include<stdio.h>
#include<conio.h>
void main()
{
	int i,j,n,m,k,temp=0,mat1[10][10],mat2[10][10],rest[10][10];
	printf("Enter The Matrix : ");
	scanf("%d %d",&n,&m);
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			printf("Enter The (%d,%d) of First matrix : ",i,j);
			scanf("%d",&mat1[i][j]);
		}
	}
	for(i=0;i<m;i++)
	{
		for(j=0;j<n;j++)
		{
			printf("Enter The (%d,%d) of Second matrix : ",i,j);
			scanf("%d",&mat2[i][j]);
		}
	}
	printf("The First Matrix\n");
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			printf("%3d",mat1[i][j]);
		}
		printf("\n");
	}
	printf("The Second Matrix\n");
	for(i=0;i<m;i++)
	{
		for(j=0;j<n;j++)
		{
			printf("%3d",mat2[i][j]);
		}
		printf("\n");
	}
	printf("The Multiplication Of Both Matrix\n");
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			for(k=0;k<m;k++)
			{
				temp+=mat1[i][k]*mat2[k][j];
			}
			rest[i][j]=temp;
			temp=0;
			printf("%3d",rest[i][j]);
		}
		printf("\n");
	}
	getch();
}
