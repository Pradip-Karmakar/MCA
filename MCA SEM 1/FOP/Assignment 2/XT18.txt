#include<stdio.h>
#include<conio.h>
void main()
{
	int i = 0, count = 0;
	char name[20];
	printf("Enter the String : ");
	scanf("%s",name);
	while(name[i] != '\0')
	{
		count++;
		i++;
	}	
	for(i = count - 1; i >= 0 ; i--)
	{
		printf("%c",name[i]);
	}
	getch();
}
