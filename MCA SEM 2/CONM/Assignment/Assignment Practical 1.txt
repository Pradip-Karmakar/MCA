#include<stdio.h>
#include<conio.h>
#include<math.h>

double fun(double x)
{
	double funx;
	funx = (x*x)-5;
	return funx;
}

double derfun(double x)
{
	double funx;
	funx = 2*x;
	return funx;
}

void cntnt(double a)
{
	double c;
	int no=0;

	printf("Enter Value:");
	scanf("%lf",&a);

	c = a-((fun(a))/(derfun(a)));
	printf("\nNO\tXn\t\tf(Xn)\t\tf'(Xn)\t\tXn+1\n");
	printf("\n------------------------------------------------------------------------------------------------------");
	
	while(fabs(fun(a))>0.0000005)
	{
		printf("\n%d\t%lf\t%lf\t%lf\t%lf\n",++no,a,fun(a),derfun(a),c);
		a = c;

		c = a-((fun(a))/(derfun(a)));
	}
	printf("\n%d\t%lf\t%lf\t%lf\t%lf\n",++no,a,fun(a),derfun(a),c);
}
